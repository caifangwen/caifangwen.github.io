// ============================================================ // useContext + createContext — 全局主题管理 // ============================================================ const ThemeContext = createContext();

// ============================================================ // useReducer — 复杂状态管理（任务的增删改查） // ============================================================ const initialTasks = [ { id: 1, text: “学习 React Hooks”, done: true, priority: “high” }, { id: 2, text: “构建任务管理应用”, done: false, priority: “high” }, { id: 3, text: “阅读官方文档”, done: false, priority: “medium” }, { id: 4, text: “写单元测试”, done: false, priority: “low” }, ];

function taskReducer(state, action) { switch (action.type) { case “ADD”: return […state, { id: Date.now(), text: action.text, done: false, priority: action.priority }]; case “TOGGLE”: return state.map(t => t.id === action.id ? { …t, done: !t.done } : t); case “DELETE”: return state.filter(t => t.id !== action.id); case “CLEAR_DONE”: return state.filter(t => !t.done); default: return state; } }

// ============================================================ // 自定义 Hook — useLocalStorage（内含 useDebugValue） // ============================================================ function useLocalStorage(key, initial) { const [value, setValue] = useState(() => { try { const stored = localStorage.getItem(key); return stored ? JSON.parse(stored) : initial; } catch { return initial; } });

// useDebugValue：在 DevTools 中显示调试信息 useDebugValue(value, v => ${key}: ${JSON.stringify(v).slice(0, 30)});

const set = useCallback((val) => { setValue(val); localStorage.setItem(key, JSON.stringify(val)); }, [key]);

return [value, set]; }
